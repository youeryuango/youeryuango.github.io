Golang中md5摘要的两种方案

  1. 文档地址 https://studygolang.com/pkgdoc
  2. 包名import "crypto/md5"
  3. 功能概述 md5包实现了MD5哈希算法
func Sum(data []byte) [Size]byte //返回数据data的MD5校验和

func New() hash.Hash // 返回一个新的使用MD5校验的hash.Hash接口

package main

import (
	"crypto/md5"
	"fmt"
	"io"
)

func main() {
	// 方案一
	str := "123456"
	data := []byte(str)
	hashStr := md5.Sum(data)
	md5Str := fmt.Sprintf("%x", hashStr)
	fmt.Println(md5Str)
	// e10adc3949ba59abbe56e057f20f883e

	// 方案二
	w := md5.New()
	io.WriteString(w, str)
	newMd5Str := fmt.Sprintf("%x",  w.Sum(nil))
	fmt.Println(newMd5Str)
	// e10adc3949ba59abbe56e057f20f883e
}
